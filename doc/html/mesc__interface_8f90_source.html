14SUBROUTINE vmic_param_constant(kinetics,micpxdef,micpdef,micparam,zse)
15 use mic_constant
16 use mic_variable
17 implicit none
18 TYPE(mic_param_xscale), INTENT(IN) :: micpxdef
19 TYPE(mic_param_default), INTENT(IN) :: micpdef
20 TYPE(mic_parameter), INTENT(INout) :: micparam
21 real(r_2) zse(ms)
22 !local variables
23 real(r_2), dimension(:,:), allocatable :: froot
24 real(r_2), dimension(:), allocatable :: totroot
25 integer nopt,npft,np,ns,kinetics
26 real(r_2) depths1,depths2,krootx
27
28
29 allocate(froot(mpft,ms))
30 allocate(totroot(mpft))
31
32 do np=1,mp
33! print *, 'np pft', np,micparam%pft(np)
34 nopt=micparam%bgctype(np)
35 do ns=1,ms
36 micparam%Q1(np,ns)=micpdef%Q1
37 micparam%Q2(np,ns)=micpdef%Q2
38 micparam%fm(np,ns)=micpdef%fm * micpxdef%xfm(nopt)
39 micparam%fs(np,ns)=micpdef%fs * micpxdef%xfs(nopt)
40 enddo !ns
41 enddo ! np
42
43 depths1=0.0;depths2=0.0
44 do ns=1,ms
45 depths2 = depths2 + zse(ns)
46 do npft=1,mpft
47 krootx = micpdef%rootbeta * micpxdef%xrootbeta(npft)
48 froot(npft,ns) = (1.0/krootx) *( exp(-krootx*depths1)-exp(-krootx*depths2))
49 enddo !npft
50 depths1=depths2
51 enddo !ns
52
53
54 do npft=1,mpft
55 totroot(npft) =sum(froot(npft,1:ms))
56 enddo
57
58 ! !normalizing
59 do ns=1,ms
60 do npft=1,mpft
61 froot(npft,ns) = froot(npft,ns)/totroot(npft)
62 enddo
63 enddo
64
65 ! calculate mp by ms all parameter values
66 do np=1, mp
67 npft=min(mpft,max(1,micparam%pft(np)))
68 nopt=micparam%bgctype(np)
69 do ns=1,ms
70 micparam%sdepth(np,ns) = zse(ns)
71 micparam%fracroot(np,ns) = froot(npft,ns)
72 enddo !"ns"
73 micparam%diffsocx(np) = micpxdef%xdiffsoc(nopt) * micpdef%diffsoc !"diffsoc" from mic_constant
74 enddo ! "np=1,mp"
75
76 if(diag==1) then
77 print *, micparam%fracroot(outp,:)
78 print *, micparam%sdepth(outp,:)
79 print *, micparam%diffsocx(outp)
80 endif
81
82 ! the following parameters are specific to kinetics3
83 if(kinetics==3) then
84 do np=1,mp
85 do ns=1,ms
86 nopt=micparam%bgctype(np)
87 micparam%kadsorp(np,ns) = micpdef%kadsorpx
88 micparam%kdesorp(np,ns) = micparam%kadsorp(np,ns) /(micpdef%kbax * micpxdef%xkba(nopt))
89 micparam%fp2a(np,ns) = micpdef%fp2ax * micpxdef%xfp2ax(nopt)
90 micparam%tvcpool(np,ns) = micpdef%tvcpoolx * micpxdef%xtvc(nopt)
91 micparam%tvppool(np,ns) = micpdef%tvppoolx * micpxdef%xtvp(nopt)
92 micparam%tvac(np,ns) = micpdef%tvacx * micpxdef%xtvac(nopt)
93 micparam%qmaxcoeff(np,ns)= micpdef%qmaxcoeff * micpxdef%xqmaxcoeff(nopt)
94 enddo !ns
95 enddo !np
96 endif ! kinetics
97
98
99 deallocate(froot)
100 deallocate(totroot)
101END SUBROUTINE vmic_param_constant

502 +(1.0-0.5* coeffa*(1.0+coeffb))*xpool(j) &
503 + 0.5* coeffa * xpool(j+1) &
504 + fluxsoc(j) *delt
505 endif

511 ct(ms) = 0.0

529 cleachloss = max(0.0,min(cleachloss,miccpool%cpool(np,ms,7)))
