770 subroutine bioturb(ndelt,ms,zse,delt,diffsocxx,fluxsoc,xpooli,xpoole)
771 ! multi-layered soil BGC including DOC and bioturbation using microbially-based BGC modeling
772 ! step 1: litter-C and SOC bioturbation treated as a diffusion process
773 ! step 2: advection of DOC along with water flux
774 ! solve dc/dt=Dd2c/dx +F(z) where c is total SOC concentration in each soil layer
775 ! bioturbation diffusion rate
776 ! boundary conditions: at the top -Ddc/dx = F0+F(1) at x=0
777 ! at the bottom: dC/dx=0 at x=h
778 ! using the fully implicit method together with Thomas method
779 ! unit for pool: mgc/cm3 (=kg C/m3)
780 ! for flux: mgc/cm3/delt (=kg c/m3/delt): g/m2/delt = 0.1 mg/cm2/delt
781 ! for length: cm
782 ! for diffsion coefficient: cm2/delt
783 use mic_constant, ONLY : r_2
784 implicit none
785 integer ndelt,ms
786 real(r_2), dimension(ms) :: zse
787 real(r_2) delt,diffsocxx
788 real(r_2), dimension(ms) :: xpooli,xpoole,xpool,fluxsoc
789 ! local variables
790 integer i,j
791 real(r_2) deltD,tot0, tot1, totflux
792 real(r_2), dimension(ms) :: xzse
793 real(r_2), dimension(ms+1) :: sdepthx
794 real(r_2) coeffA, coeffB
795 real(r_2), dimension(ms) :: at,bt,ct,rt
796
797 ! calculate the mid-point of each layer
798 sdepthx(1) = 0.0 ! depth of a layer from the top (x_0.5=0.0 eg soil surface)
799 do j=2,ms+1
800 sdepthx(j) = sdepthx(j-1) + zse(j-1)*100.0 ! depth of the bottom of each layer (eg x_j+0.5)
801 !*100 to convert from m to cm
802 enddo
803
804 do j=1,ms
805 xzse(j) = 0.5 * (sdepthx(j) + sdepthx(j+1)) ! depth of midpoint of a layer j (x_j)
806 enddo
807
808 deltd = diffsocxx * delt
809
810 xpool = xpooli
811 tot0 = 0.0
812 do j=1,ms
813 tot0 = tot0 + xpool(j) * zse(j)*100.0 !*100 convert m to cm
814 enddo
815
816 do i=1,ndelt
817 do j=1,ms
818 if(j==1) then
819 coeffb = 1.0/(sdepthx(2)-sdepthx(1))
820 coeffa = deltd*coeffb/(xzse(2)-xzse(1))
821 ! Crank-Nicholson
822 at(1) = 0.0
823 bt(1) = 1.0 + 0.5 * coeffa
824 ct(1) = - 0.5 * coeffa
825 rt(1) = (1.0-0.5*coeffa) * xpool(1) + 0.5 * coeffa * xpool(2) &
826 + fluxsoc(1) * delt
827 endif
828 if(j>1.and.j<ms) then
829 coeffa = deltd/((xzse(j+1)-xzse(j))*(sdepthx(j+1)-sdepthx(j)))
830 coeffb = (xzse(j+1)-xzse(j))/(xzse(j)-xzse(j-1))
831 ! Crank-Nicholson
832 at(j) = -0.5 * coeffa * coeffb
833 bt(j) = 1.0+0.5 * coeffa *(1.0+coeffb)
834 ct(j) = -0.5 * coeffa
835 rt(j) = 0.5 * coeffa * coeffb * xpool(j-1) &
836 +(1.0-0.5* coeffa*(1.0+coeffb))*xpool(j) &
837 + 0.5* coeffa * xpool(j+1) &
838 + fluxsoc(j) *delt
839 endif
840 if(j==ms) then
841 coeffa = deltd/((xzse(ms)-xzse(ms-1))*(sdepthx(ms+1) - sdepthx(ms)))
842 ! Crank-Nicholson
843 at(ms) = -0.5 * coeffa
844 bt(ms) = 1.0 + 0.5 * coeffa
845 ct(ms) = 0.0
846 rt(ms) = 0.5* coeffa * xpool(ms-1) + (1.0-0.5 * coeffa) * xpool(ms) &
847 + fluxsoc(ms) * delt
848 endif
849 enddo
850
851 call tridag(at,bt,ct,rt,xpool,ms)
852 enddo
853 xpoole = xpool
854
855 tot1 = 0.0
856 totflux=0.0
857 do j=1,ms
858 tot1 = tot1 + xpool(j) * zse(j) *100.0
859 totflux = totflux + fluxsoc(j) * zse(j) *100.0

864 subroutine tridag(at,bt,ct,rt,u,ms)
865 ! solving the triadigonal matrix (numerical recipes, p43)
866 ! linear equation: A* u(i-1) + B *u(i) + C * u(i+1) = R,
867 ! where i is soil layer, u(i-1), u(i) and u(i+1) are at time step t
868 ! NOTE: bt(1) should not be equal to 0.0, otherwise rewrite the equation
869 use mic_constant, ONLY : r_2
870 implicit none
871 integer, parameter :: nmax=500
872 integer ms
873 real(r_2), dimension(ms) :: at,bt,ct,rt,u
874 integer j
875 real(r_2) bet
876 real(r_2), dimension(nmax) :: gam
877
878 bet = bt(1)
879 u(1) = rt(1)/bet
880 do j=2,ms
881 gam(j) = ct(j-1)/bet
882 bet = bt(j)-at(j)* gam(j)
883 if(bet ==0) then
884 print *, 'triag failed'
885 stop
886 endif
887 u(j) = (rt(j) - at(j) * u(j-1))/bet
888 enddo
889 do j=ms-1,1,-1
890 u(j) = u(j) -gam(j+1) * u(j+1)
891 enddo
892 end subroutine tridag
